## ğŸ”§ TECHNICAL TIPS FOR EACH FEATURE

### Performance
- Use `will-change` only on animated elements
- Lazy load components with dynamic import
- Optimize images (use WebP with fallback)
- Throttle scroll events
- Use CSS transforms instead of position changes

### Animations
- Keep animations under 1s for UI elements
- Use `duration` of 2-3s for entrance animations
- Always include `exit` animations (not just enter)
- Use `type: "spring"` for organic feel
- Stagger effects with `transition={{ delay: index * 0.1 }}`

### User Experience
- Always allow skipping animations
- Respect `prefers-reduced-motion`
- Add loading states
- Test on slow devices
- Include accessibility features (ARIA labels)

const StatCounter = ({ label, value, emoji }: { label: string; value: string; emoji: string }) => {
  const ref = useRef(null)
  const isInView = useInView(ref, { once: true })
  const [displayValue, setDisplayValue] = useState(0)

  useEffect(() => {
    if (!isInView) return

    let timeout: NodeJS.Timeout
    if (value.includes('+')) {
      const num = parseInt(value.replace('+', ''))
      const duration = 2000
      const steps = 60
      const increment = num / steps

      let current = 0
      const interval = setInterval(() => {
        current += increment
        if (current >= num) {
          setDisplayValue(num)
          clearInterval(interval)
        } else {
          setDisplayValue(Math.floor(current))
        }
      }, duration / steps)

      return () => clearInterval(interval)
    }
  }, [isInView, value])

  return (
    <motion.div
      ref={ref}
      className="glass-effect rounded-2xl p-8 text-center"
      whileHover={{ scale: 1.05, y: -10 }}
      initial={{ opacity: 0, y: 30 }}
      whileInView={{ opacity: 1, y: 0 }}
      viewport={{ once: true }}
    >
      <motion.div
        className="text-5xl mb-4"
        animate={{ rotate: [0, 10, -10, 0] }}
        transition={{ duration: 3, repeat: Infinity }}
      >
        {emoji}
      </motion.div>
      <motion.p className="font-great text-4xl text-pink-700 mb-2">
        {value.includes('+') ? `${displayValue}+` : value}
      </motion.p>
      <p className="font-dancing text-lg text-pink-600">{label}</p>
    </motion.div>
  )
}
